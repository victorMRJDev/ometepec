CREATE TABLE `licencias` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nombres` varchar(255) NOT NULL,
  `apellidoPaterno` varchar(255) NOT NULL,
  `apellidoMaterno` varchar(255) DEFAULT NULL,
  `telefono` varchar(20) DEFAULT NULL,
  `sexo` enum('masculino','femenino','otro') NOT NULL,
  `fechaNacimiento` date DEFAULT NULL,
  `curp` varchar(18) NOT NULL,
  `rfc` varchar(13) DEFAULT NULL,
  `tipoSangre` varchar(3) DEFAULT NULL,
  `alergias` text,
  `domicilio` text,
  `contactoEmergencia` varchar(20) DEFAULT NULL,
  `donadorOrganos` enum('si','no') DEFAULT 'no',
  `restricMedica` text,
  `status` varchar(255) DEFAULT NULL,
  `fotografia` varchar(255) DEFAULT NULL,
  `created_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `firma` varchar(255) DEFAULT NULL,
  `fechaExpedicion` date DEFAULT NULL,
  `fechaVencimiento` date DEFAULT NULL,
  `licenciaDe` varchar(255) DEFAULT NULL,
  `tipoLicencia` varchar(255) DEFAULT NULL,
  `numLicencia` varchar(255) DEFAULT NULL,
  `idPersona` int DEFAULT NULL,
  `costo` varchar(50) DEFAULT NULL,
  `tipoTramite` varchar(50) DEFAULT NULL,
  `duracion` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `curp` (`curp`),
  UNIQUE KEY `unique_numLicencia` (`numLicencia`)
);

CREATE TABLE `placas` (
  `id` int NOT NULL AUTO_INCREMENT,
  `folioUnico` varchar(50) NOT NULL,
  `fechaExpedicion` date NOT NULL,
  `fechaVencimiento` date NOT NULL,
  `numSerie` varchar(100) DEFAULT NULL,
  `numMotor` varchar(100) DEFAULT NULL,
  `marca` varchar(100) DEFAULT NULL,
  `yearModelo` int DEFAULT NULL,
  `color` varchar(50) DEFAULT NULL,
  `rfcSolicitante` varchar(20) DEFAULT NULL,
  `domicilio` varchar(255) DEFAULT NULL,
  `nombres` varchar(255) DEFAULT NULL,
  `apellidoPaterno` varchar(255) DEFAULT NULL,
  `apellidoMaterno` varchar(255) DEFAULT NULL,
  `importe` decimal(10,2) DEFAULT NULL,
  `status` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`)
);

CREATE TABLE `secuencias` (
  `nombre` varchar(50) NOT NULL,
  `valor` int NOT NULL,
  PRIMARY KEY (`nombre`)
);

CREATE TABLE `usuarios` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nombre` varchar(100) NOT NULL,
  `apellidoPaterno` varchar(100) NOT NULL,
  `apellidoMaterno` varchar(100) NOT NULL,
  `curp` varchar(18) NOT NULL,
  `numEmpleado` varchar(20) NOT NULL,
  `rfc` varchar(13) NOT NULL,
  `edad` int NOT NULL,
  `domicilio` varchar(255) DEFAULT NULL,
  `genero` varchar(100) NOT NULL,
  `rolUsuario` enum('administrador','usuario') NOT NULL,
  `correo` varchar(255) NOT NULL,
  `password` varchar(255) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `curp` (`curp`),
  UNIQUE KEY `numEmpleado` (`numEmpleado`),
  UNIQUE KEY `rfc` (`rfc`)
);
